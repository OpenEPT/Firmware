94 static void prvNETWORK_Task()
95 {
96  memset(&prvNETWORK_MAC_CONFIG, 0, sizeof(ETH_MACConfigTypeDef));
97  int32_t PHYLinkState = 0;
98  uint32_t linkchanged = 0U, speed = 0U, duplex = 0U;

113 
114  /* add the network interface (IPv4/IPv6) with RTOS */
115  netif_add(&prvNETWORK_DATA.gnetif, &prvNETWORK_DATA.ipaddr, &prvNETWORK_DATA.netmask, &prvNETWORK_DATA.gw, NULL, &ethernetif_init, &tcpip_input);
116 
117  LOGGING_Write("Network", LOGGING_MSG_TYPE_INFO, "Network interface added - Info:\r\n");
118  LOGGING_Write("Network", LOGGING_MSG_TYPE_INFO, " - IP : %s\r\n", NETWORK_DEVICE_IP_ADDRESS);
119  LOGGING_Write("Network", LOGGING_MSG_TYPE_INFO, " - MASK : %s\r\n", NETWORK_DEVICE_IP_MASK);
120  LOGGING_Write("Network", LOGGING_MSG_TYPE_INFO, " - Gateway : %s\r\n", NETWORK_DEVICE_IP_GW);

122  /* Registers the default network interface */

128  netif_set_up(&prvNETWORK_DATA.gnetif);

172  case LAN8742_STATUS_10MBITS_FULLDUPLEX:
173  duplex = ETH_FULLDUPLEX_MODE;
174  speed = ETH_SPEED_10M;
175  linkchanged = 1;
176  break;
177  case LAN8742_STATUS_10MBITS_HALFDUPLEX:
178  duplex = ETH_HALFDUPLEX_MODE;
179  speed = ETH_SPEED_10M;
180  linkchanged = 1;
181  break;
182  default:
183  break;
184  }

188  /* Get MAC Config MAC */
189  HAL_ETH_GetMACConfig(&HETH, &prvNETWORK_MAC_CONFIG);
190  prvNETWORK_MAC_CONFIG.DuplexMode = duplex;
191  prvNETWORK_MAC_CONFIG.Speed = speed;
192  HAL_ETH_SetMACConfig(&HETH, &prvNETWORK_MAC_CONFIG);
193  HAL_ETH_Start_IT(&HETH);
