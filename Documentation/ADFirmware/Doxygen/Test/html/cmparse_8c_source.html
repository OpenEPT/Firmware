131 {
132  char* tmpArgBuffPtr = argBuffer;
133  uint32_t tmpArgBufferSizeProcessed = 0;
134  if(*argBufferSize == 0) return NULL;
135  key->size = 0;
136  value->size = 0;
137  //arguments are in format -key=value
138  while(*tmpArgBuffPtr != '-'){
139  tmpArgBufferSizeProcessed += 1;
140  tmpArgBuffPtr += 1;
141  if(tmpArgBufferSizeProcessed == *argBufferSize) return NULL;
142  }
143  tmpArgBufferSizeProcessed += 1;
144  tmpArgBuffPtr += 1;
145  while(*tmpArgBuffPtr != '=')
146  {
147  key->value[key->size] = *tmpArgBuffPtr;
148  key->size += 1;
149  tmpArgBufferSizeProcessed += 1;
150  tmpArgBuffPtr += 1;
151  if(tmpArgBufferSizeProcessed == *argBufferSize) return NULL;
152  }
153  tmpArgBufferSizeProcessed += 1;
154  tmpArgBuffPtr += 1;
155  while(*tmpArgBuffPtr != ' ' && tmpArgBufferSizeProcessed < *argBufferSize)
156  {
157  value->value[value->size] = *tmpArgBuffPtr;
158  value->size += 1;
159  tmpArgBufferSizeProcessed += 1;
160  tmpArgBuffPtr += 1;
161  }
