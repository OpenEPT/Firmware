380 static ads9224r_status_t prvADS9224R_CONF_SPI_Master_ReadReg(uint8_t reg, uint8_t* data, uint32_t timeout)
381 {
382  uint8_t txData[2];
383 
384  /* Prepare data */
385  txData[0] = 0x20 | (reg & 0x0F);
386  txData[1] = 0x00;
387 
388  /*CS go low*/
389  DRV_GPIO_Pin_SetState(ADS9224R_CS_PORT, ADS9224R_CS_PIN, DRV_GPIO_PIN_STATE_RESET);
390 
391  /*Transmit command*/
392  if(DRV_SPI_TransmitData(txData, 2, timeout) != DRV_SPI_STATUS_OK) return DRV_SPI_STATUS_ERROR;
393 
394  /*CS go high*/
395  DRV_GPIO_Pin_SetState(ADS9224R_CS_PORT, ADS9224R_CS_PIN, DRV_GPIO_PIN_STATE_SET);
396 
397  /* Wait for ready signal */
398  while(DRV_GPIO_Pin_ReadState(ADS9224R_READY_STROBE_PORT, ADS9224R_READY_STROBE_PIN) != DRV_GPIO_PIN_STATE_SET);
399 
400  /* CS go low */
401  DRV_GPIO_Pin_SetState(ADS9224R_CS_PORT, ADS9224R_CS_PIN, DRV_GPIO_PIN_STATE_RESET);
402 
403  /*Receive data*/
404  if(DRV_SPI_ReceiveData(data, 1, timeout) != DRV_SPI_STATUS_OK) return DRV_SPI_STATUS_ERROR;
405 
406  /*CS go high*/
407  DRV_GPIO_Pin_SetState(ADS9224R_CS_PORT, ADS9224R_CS_PIN, DRV_GPIO_PIN_STATE_SET);
408 
409  return ADS9224R_STATUS_OK;
410 }
