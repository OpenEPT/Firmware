54 /*
55 @Note: This interface is implemented to operate in zero-copy mode only:
56  - Rx Buffers will be allocated from LwIP stack Rx memory pool,
57  then passed to ETH HAL driver.
58  - Tx Buffers will be allocated from LwIP stack memory heap,
59  then passed to ETH HAL driver.
60 
61 @Notes:
62  1.a. ETH DMA Rx descriptors must be contiguous, the default count is 4,
63  to customize it please redefine ETH_RX_DESC_CNT in ETH GUI (Rx Descriptor Length)
64  so that updated value will be generated in stm32xxxx_hal_conf.h
65  1.b. ETH DMA Tx descriptors must be contiguous, the default count is 4,
66  to customize it please redefine ETH_TX_DESC_CNT in ETH GUI (Tx Descriptor Length)
67  so that updated value will be generated in stm32xxxx_hal_conf.h
68

70  2.b. Rx Buffers must have the same size: ETH_RX_BUFFER_SIZE, this value must
71  passed to ETH DMA in the init field (heth.Init.RxBuffLen)

286 static err_t low_level_output(struct netif *netif, struct pbuf *p)
287 {
288  uint32_t i = 0U;
289  struct pbuf *q = NULL;
290  err_t errval = ERR_OK;
291  ETH_BufferTypeDef Txbuffer[ETH_TX_DESC_CNT];
292 
293  memset(Txbuffer, 0 , ETH_TX_DESC_CNT*sizeof(ETH_BufferTypeDef));
294 
295  for(q = p; q != NULL; q = q->next)
296  {
297  if(i >= ETH_TX_DESC_CNT) return ERR_IF;
298 
299  Txbuffer[i].buffer = q->payload;
300  Txbuffer[i].len = q->len;
301 
302  if(i>0)
303  {
304  Txbuffer[i-1].next = &Txbuffer[i];
305  }
306 
307  if(q->next == NULL)
308  {
309  Txbuffer[i].next = NULL;
310  }
311 
312  i++;
313  }
314 
315  TxConfig.Length = p->tot_len;
316  TxConfig.TxBuffer = Txbuffer;
317  TxConfig.pData = p;
318 
319  pbuf_ref(p);
320 
321  do
322  {
323  if(HAL_ETH_Transmit_IT(&HETH, &TxConfig) == HAL_OK)
324  {
325  errval = ERR_OK;
326  }
327  else
328  {
329 
330  if(HAL_ETH_GetError(&HETH) & HAL_ETH_ERROR_BUSY)
331  {
332  /* Wait for descriptors to become available */
333  osSemaphoreAcquire( TxPktSemaphore, ETHIF_TX_TIMEOUT);
334  HAL_ETH_ReleaseTxPacket(&HETH);
335  errval = ERR_BUF;
336  }
337  else
338  {
339  /* Other error */
340  pbuf_free(p);
341  errval = ERR_IF;
342  }
343  }
344  }while(errval == ERR_BUF);

716 void HAL_ETH_RxLinkCallback(void **pStart, void **pEnd, uint8_t *buff, uint16_t Length)
717 {
718  struct pbuf **ppStart = (struct pbuf **)pStart;
719  struct pbuf **ppEnd = (struct pbuf **)pEnd;
720  struct pbuf *p = NULL;
721 
722  /* Get the struct pbuf from the buff address. */
723  p = (struct pbuf *)(buff - offsetof(RxBuff_t, buff));
724  p->next = NULL;
725  p->tot_len = 0;
726  p->len = Length;
727 
728  /* Chain the buffer. */
729  if (!*ppStart)
730  {
731  /* The first buffer of the packet. */
732  *ppStart = p;
733  }
734  else
735  {
736  /* Chain the buffer to the end of the packet. */
737  (*ppEnd)->next = p;
738  }
739 
740  *ppEnd = p;
741 
742  /* Update the total length of all the buffers of the chain. Each pbuf in the chain should have its tot_len
743  * set to its own length, plus the length of all the following pbufs in the chain. */
744  for (p = *ppStart; p != NULL; p = p->next)
745  {
746  p->tot_len += Length;
747  }
748 
749  /* Invalidate data cache because Rx DMA's writing to physical memory makes it stale. */
750  SCB_InvalidateDCache_by_Addr((uint32_t *)buff, Length);
751 }
